 1. Clone and run locally on laptop

  # clone
  git clone <your-repo-url> hoko
  cd hoko

  # backend
  cd server
  npm install
  npm run dev   # or npm start (check package.json)
  # backend should run on http://127.0.0.1:5000

  # frontend (new terminal)
  cd ../client
  npm install
  npm run dev   # local frontend URL shown in terminal

  2. Push laptop changes and deploy on host

  # on laptop
  git add .
  git commit -m "your change message"
  git push origin main   # or push branch + merge PR

  # on host server
  cd /var/www/hoko
  git pull origin main

  cd /var/www/hoko/server
  npm install
  pm2 restart all   # or your backend service restart

  cd /var/www/hoko/client
  npm install
  npm run build
